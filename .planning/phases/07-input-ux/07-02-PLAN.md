       const handleKeyIntercept = useCallback((input: string, key: Key, setText: (t: string) => void): boolean => {
         if (!showSuggestions) return false;
         const filtered = filterCommands(currentInput);
         if (key.upArrow) {
           setSuggestionIndex(prev => Math.max(0, prev - 1));
           return true;
         }
         if (key.downArrow) {
           setSuggestionIndex(prev => Math.min(filtered.length - 1, prev + 1));
           return true;
         }
         if (key.return && filtered.length > 0) {
           const selected = filtered[suggestionIndex];
           if (selected) {
             setText(selected.name + ' ');  // 명령어 + 공백
             setShowSuggestions(false);
           }
           return true;
         }
         if (key.escape) {
           setShowSuggestions(false);
           return true;
         }
         return false;
       }, [showSuggestions, currentInput, suggestionIndex]);
